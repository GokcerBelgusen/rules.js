 31         var Class = (function() {
 32                          var __extending = {};
 33 
 34                          return {
 35                              extend: function(parent, def) {
 36                                  if (arguments.length == 1) { def = parent; parent = null; }
 37                                  var func = function() {
 38                                      if (arguments[0] ==  __extending) { return; }
 39                                      this.initialize.apply(this, arguments);
 40                                  };
 41                                  if (typeof(parent) == 'function') {
 42                                      func.prototype = new parent( __extending);
 43                                  }
 44                                  var mixins = [];
 45                                  if (def && def.include) {
 46                                      if (def.include.reverse) {
 47                                          // methods defined in later mixins should override prior
 48                                          mixins = mixins.concat(def.include.reverse());
 49                                      } else {
 50                                          mixins.push(def.include);
 51                                      }
 52                                      delete def.include; // clean syntax sugar
 53                                  }
 54                                  if (def) Class.inherit(func.prototype, def);
 55                                  for (var i = 0; (mixin = mixins[i]); i++) {
 56                                      Class.mixin(func.prototype, mixin);
 57                                  }
 58                                  return func;
 59                              },
 60                              mixin: function (dest, src, clobber) {
 61                                  clobber = clobber || false;
 62                                  if (typeof(src) != 'undefined' && src !== null) {
 63                                      for (var prop in src) {
 64                                          if (clobber || (!dest[prop] && typeof(src[prop]) == 'function')) {
 65                                              dest[prop] = src[prop];
 66                                          }
 67                                      }
 68                                  }
 69                                  return dest;
 70                              },
 71                              inherit: function(dest, src, fname) {
 72                                  if (arguments.length == 3) {
 73                                      var ancestor = dest[fname], descendent = src[fname], method = descendent;
 74                                      descendent = function() {
 75                                          var ref = this.parent; this.parent = ancestor;
 76                                          var result = method.apply(this, arguments);
 77                                          ref ? this.parent = ref : delete this.parent;
 78                                          return result;
 79                                      };
 80                                      // mask the underlying method
 81                                      descendent.valueOf = function() { return method; };
 82                                      descendent.toString = function() { return method.toString(); };
 83                                      dest[fname] = descendent;
 84                                  } else {
 85                                      for (var prop in src) {
 86                                          if (dest[prop] && typeof(src[prop]) == 'function') {
 87                                              Class.inherit(dest, src, prop);
 88                                          } else {
 89                                              dest[prop] = src[prop];
 90                                          }
 91                                      }
 92                                  }
 93                                  return dest;
 94                              },
 95                              singleton: function() {
 96                                  var args = arguments;
 97                                  if (args.length == 2 && args[0].getInstance) {
 98                                      var klass = args[0].getInstance(__extending);
 99                                      // we're extending a singleton swap it out for it's class
100                                      if (klass) { args[0] = klass; }
101                                  }
102 
103                                  return (function(args){
104                                              // store instance and class in private variables
105                                              var instance = false;
106                                              var klass = Class.extend.apply(args.callee, args);
107                                              return {
108                                                  getInstance: function () {
109                                                      if (arguments[0] == __extending) return klass;
110                                                      if (instance) return instance;
111                                                      return (instance = new klass());
112                                                  }
113                                              };
114                                          })(args);
115                              }
116                          };
117                      })();

164         // Very dirty utility to look for specified children in path
165         // Should be replaced by more efficient cross browser/javascript runtine XML XPATH like utility
166         function getMatchingTags(data, path, array) {
167             //console.log('Looking for path = ' , path, ' @' , data.tagName);
168             if (path === "") {
169                 return;
170             }
171 
172             var index = path.indexOf("/");
173             var children = data.childNodes;
174             var i;
175             var son;
176             if (path.charAt(0) === '*') {
177                 for (i = 0; i < children.length; i++) {
178                     son = children.item(i);
179                     if (son.nodeType !== 1) {
180                         continue;
181                     }
182                     if (index === -1) {
183                         array.push(son);
184                     }
185                     if (index > 0) {
186                         getMatchingTags(son, path.substr(index + 1), array);
187                     }
188                     if (index > 1) {
189                         getMatchingTags(son, path, array);
190                     }
191                 }
192             } else {
193                 var tag;
194                 if (index !== -1) {
195                     tag = path.substr(0, index);
196                 } else {
197                     tag = path;
198                 }
199                 //console.log('Looking for tag = ' + tag);
200                 children = data.childNodes;
201                 for (i = 0; i < children.length; i++) {
202                     son = children.item(i);
203                     if (son.nodeType !== 1) {
204                         continue;
205                     }
206                     if (son.tagName !== tag){
207                         continue;
208                     }
209                     if (index === -1) {
210                         array.push(son);
211                     }
212                     if (index > 0) {
213                         getMatchingTags(son, path.substr(index + 1), array);
214                     }
215                 }
216             }
217         }

249              * @class
250              * The Rule Context class
251              * <br/>
252              * <br/>
253              * The Rule Context is the way through which the engine holds grip on objects
254              * on which it performs its reasoning. It holds also the required cache for efficient
255              * data access.
256              * <br/>
257              * <br/>
258              * This class is not directly exposed in the API. It is used internally by the engine.
259              * A rule context is allocated for each call of the engine and holds the
260              * grip on all variables on which the engine will apply rules
261              * <br/>
262              * <br/>
263              * For more advanced uses, one may need to access and manipulate this

439              * Internal function that manually parses the access path to an object
440              * @private
441              */
442             _getPathObject : function(path, create) {
443                 var object = this.values;
444                 var i = 0;
445                 var j = path.indexOf(".");
446                 var father = this.values;
447                 while (j != -1) {
448                     var attribute = path.substring(i,j);
449                     object = father.getAttributeValue(attribute);
450                     if (!object) {
451                         if (create) {
452                             object = new RuleResult();
453                             father.setAttributeValue(attribute,object);
454                         }
455                         else {
456                             return null;
457                         }
458                     }
459                     object = this.wrapObject(path.substring(0,j), object, father);
460                     i = j+1;
461                     j = path.indexOf(".",i);
462                     father = object;
463                 }
464                 return object;

476             /**
477              *  Internal function to extract an expression with an offset in a value string
478              * @private
479              */
480             _getExpressionIndex : function(expression, start) {
481                 var i=start+1;
482                 var level = 0;
483                 var inString = false;
484                 while ((i < expression.length) && ((expression.charAt(i) != ')') || (level > 0))) {
485                     if ((expression.charAt(i) =='"') || (expression.charAt(i) =='\'')) {
486                         inString  = !inString;
487                         i++;
488                         continue;
489                     }
490                     if (inString) {
491                         i++;
492                         continue;
493                     }
494                     if (expression.charAt(i) =='(') {
495                         level++;
496                     }
497                     if (expression.charAt(i) ==')') {
498                         level--;
499                     }
500                     i++;
501                 }
502                 if ((i > expression.length) || (expression.charAt(i) != ')') || inString) {
503                     console.log("Syntax error in expression ", expression);
504                     return -1;
505                 }
506                 else {
507                     return i;
508                 }
509             },

511             /**
512              * Internal function to compute a value from an expression, potentially recursive
513              * @private
514              */
515             _getExpressionValue : function(expression) {
516                 var i=0;
517                 var args = new Array();
518                 var from = 0;
519                 var inString = false;
520                 while (i < expression.length) {
521                     if ((expression.charAt(i) =='"') || (expression.charAt(i) =='\'')) {
522                         inString  = !inString;
523                         if (inString) {
524                             from=i; i++;
525                             continue;
526                         }
527                         else {
528                             args.push(expression.substring(from,i+1));
529                             i++; from=i;
530                             continue;
531                         }
532                     }
533                     if (inString) {
534                         i++;
535                         continue;
536                     }
537                     if ((expression.charAt(i) == ' ') || (expression.charAt(i) == '\t') || (expression.charAt(i) == '\n')) {
538                         if (from == i) {
539                             i++; from++;
540                         }
541                         else {
542                             args.push(expression.substring(from,i));
543                             i++; from=i;
544                         }
545                         continue;
546                     }
547                     i++;
548                 }
549 
550                 if (inString) {
551                     console.log("Error: syntax error in expression in string : ", expression);
552                     return null;
553                 }
554 
555                 if (from != i) {
556                     args.push(expression.substring(from,i));
557                 }
558                 var values = new Array();
559                 for (i=0; i<args.length; i++) {
560                     values[i] = this.getValue(args[i]);
561                 }
562 
563                 if (values.length == 0) {
564                     console.log("Error: syntax error in expression : ", expression);
565                     return null;
566                 }
567                 var funcall = values[0];
568 
569                 if (funcall == '+') funcall="plus";
570                 if (funcall == '-') funcall="minus";
571                 if (funcall == '/') funcall="div";
572                 if (funcall == '*') funcall="mul";
573 
574                 var engine = this.getEngine();
575                 var functionHandler = engine.getFunctionHandler(funcall);
576 
577                 if (functionHandler) {
578                     return functionHandler(this,values);
579                 }
580                 else {
581                     console.log("Error: unknown function in expression : ", args[0], values[0], expression);
582                     return null;
583                 }
584             },

586             /**
587              * General entry point in charge of retrieving a value from a path or an expression <br/>
588              * <br/>
589              * The general form of such expression is either <br/>
590              * - a quoted litteral (with simple or double quotes) <br/>
591              * - a string prefixed with a $ sign meaning that what follows (without spaces)
592              * is an access path in context
593              * - a string starting with a ( , then it is considered an expression in lisp
594              * like syntax with first item being a function, and following values are parameters.
595              * Expressions can be nested at will <br/>
596              * - a litteral in any other case
597              */
598             getValue : function(path) {
599                 if (!path) return null;
600                 if (path.length == 0) return null;
601                 if ((path.charAt(0) == '"') || (path.charAt(0) == '\'')) {
602                     if (path.length < 2) return null;
603                     return path.substring(1,path.length-1);
604                 }
605                 if ((path.charAt(0) == '$') && (path.length > 1) && (path.charAt(1) !='{')) {
606                     path = path.substring(1);
607                     var object = this._getPathObject(path,false);
608                     if (object) {
609                         var attribute = this._getPathAttribute(path);
610                         return object.getAttributeValue(attribute);
611                     }
612                     else {
613                         return null;
614                     }
615                 }
616                 if (path.charAt(0) == '(') {
617                     var index = this._getExpressionIndex(path, 0);
618                     if (index == -1) {
619                         console.log("Syntax error in expression ", path);
620                         return null;
621                     }
622                     else {
623                         return this._getExpressionValue(path.substring(1,index));
624                     }
625                 }
626                 return path;
627             },
628

667         var Rule = {
668             /**
669              * @class
670              * This class represents the rules objects of the engine. These objects are directly mapped
671              * against their XML source, without (for the moment) compilation to an alternative form
672              * more efficient or convenient for execution. <br/>
673              * <br/>
674              * This class is not exposed either to the client API. It is only "seen" from the client
675              * application as an XML object that customizes the behaviour of the engine in a
676              * "declarative" form. <br/>
677              *
678              * @constructs
679              * @description This this takes an XML source code rule and registers "fast" access to the conditions
680              * actions and subrules. It does propagate the rule creation to embedded rules as well.
681              * See Rule syntax in documentation for further information on rule syntax.
682              * @param ruleXML The source code for the rule as an XML object
683              * @param father The embedding father rule if any (as an object)
684              * @param id The unique id allocated to identify this rule (integer counter).
685              */
686             initialize : function(ruleXML, father, id) {
687                 console.log('Creating rule ', id , ruleXML);
688                 this.id = id;
689                 this.ruleXML = ruleXML;
690                 this.conditionsXML = new Array();
691                 getMatchingTags(ruleXML,"IF/*", this.conditionsXML);
692 
693                 this.actionsXML = new Array();
694                 getMatchingTags(ruleXML,"THEN/*", this.actionsXML);
695 
696                 this.rulesXML = new Array();
697                 getMatchingTags(ruleXML,"RULE", this.rulesXML);
698 
699                 // Small shortcut for those who don't like verbose syntax
700                 // Any element that is not a condition nor a rule nor an explicit THEN is an action
701                 var sons = ruleXML.childNodes;
702                 for (var i=0; i<sons.length; i++) {
703                     var son = sons.item(i);
704                     if (son.nodeType !== 1) {
705                         continue;
706                     }
707                     if ((son.tagName != "IF") && (son.tagName != "THEN") && (son.tagName != "RULE")) {
708                         this.actionsXML.push(son);
709                     }
710                 }
711                 this.father = father;
712             },
713 
714             /**
715              * @return A boolean indicating whether the specified rule has
716              * an "optimized" condition with this access key condition
717              * @param key The key (access path) that will be used
718              */
719             hasKey : function(key) {
720                 var hasKey = this.ruleXML.hasAttribute(key);
721                 if (hasKey)
722                     return hasKey;
723                 if (!this.father)
724                     return false;
725                 return this.father.hasKey(key);
726             },
727 
728             /**
729              * @return The key value for the optimized key condition to be met
730              * for this rule to fire. For a compiled condition, it MUST be a constant.
731              * @param key The key (access path) that will be used
732              */
733             getKey : function(key) {
734                 var hasKey = this.ruleXML.hasAttribute(key);
735                 if (hasKey)
736                     return this.ruleXML.getAttribute(key);

861             /**
862              * This is the major algorithm to build the index of all rules. <br/>
863              * The index is a recursive data structure that "points" to the matching rules.
864              */
865             addRule : function(keys, index, rule) {
866                 // Check if index tree is at a leaf
867                 if (index >= keys.length) {
868                     // If so and there is no rule in this index, create an empty array
869                     if (!this.rules) {
870                         this.rules = new Array();
871                     }
872                     // Add the rule to the leaf index
873                     this.rules.push(rule);
874                     return;
875                 }
876                 else {
877                     // Process next tree level
878                     var key = keys[index];
879                     this.key = key;
880                     // Check if this rule actually has this key as an optimized condition
881                     if (rule.hasKey(key)) {
882                         // The rule does use this key
883                         var value = rule.getKey(key);
884                         // If no subindex, then allocate it
885                         if (!this.indexes)
886                             this.indexes = new Object();
887                         // If no subindex of this key, then allocate it as well
888                         if (!this.indexes[value]) {
889                             this.indexes[value] = new RuleIndex();
890                             // Then one must reinject all rules that happened to be
891                             // under the nokey entry
892                             if (this.noKey) {
893                                 var noKeyRules = new Array();
894                                 this.noKey.getRules(null, noKeyRules);
895                                 for (var i=0; i< noKeyRules.length; i++) {
896                                     this.indexes[value].addRule(keys, index, noKeyRules[i]);
897                                 }
898                             }
899                         }
900 
901                         this.indexes[value].addRule(keys, index+1, rule);
902                     }
903                     else {
904                         // This rule does not use this key, hence all key values apply
905                         if (!this.noKey) {
906                             this.noKey = new RuleIndex();
907                         }
908                         this.noKey.addRule(keys, index+1, rule);
909                         // And add this nokey rule to any existing key rule
910                         if (this.indexes) {
911                             for (var other in this.indexes) {
912                                 this.indexes[other].addRule(keys, index+1, rule);
913                             }
914                         }
915                     }
916                 }

943                                 this.noKey.getRules(context,keys,index+1,rules);

1416                     if (keys.indexOf(attributeName) != -1) {
1417                         console.log("Using a key condition twice in a rule hierarchy is not supported :" + attributeName, keys, keyOffset);

1421                     keys.push(attributeName);

1425                     count += this._keyStatsOfRule(rulesXML[i], keys, keysStats);
1426                 }
1427 
1428                 for (i=0; i< keyCount; i++) {
1429                     var key = keys[keyOffset+i];
1430                     var value = ruleXML.getAttribute(key);
1431                     if (!keysStats[key]) {
1432                         keysStats[key] = new Object();
1433                         keysStats[key]._count = count;
1434                     } else {
1435                         keysStats[key]._count = keysStats[key]._count + count;
1436                     }
1437 
1438                     if (keysStats[key][value]) {
1439                         keysStats[key][value] = keysStats[key][value] + count;
1440                     }
1441                     else {
1442                         keysStats[key][value] = count;
1443                     }
1444                 }
1445

1447                     keys.pop();
1448                 }
1449

1463             // For the moment, only the internal run function of the engine
1464             // but may be an entry point for more advanced uses eventually
1465             run : function (context) {
1466                 context.setEngine(this);
1467                 var rules = new Array();
1468                 // First use index to get the set of rules candidate for execution
1469                 this.index.getRules(context, this.keys, 0, rules);
1470 
1471                 // Then find the highest priority rule
1472                 var maxPriority = Number.NEGATIVE_INFINITY;
1473                 for (var i = 0 ; i < rules.length; i++) {
1474                     var priority = rules[i].getPriority();
1475                     if (maxPriority < priority)
1476                         maxPriority = priority;
1477                 }
1478                 var firstMaxPriority = maxPriority;
1479 
1480                 // All priorities will be scanned in descending order (unless restart)
1481                 while (maxPriority > Number.NEGATIVE_INFINITY) {
1482                     var nextPriority = Number.NEGATIVE_INFINITY;
1483                     console.log("Processing rules of priority " + maxPriority);
1484 
1485                     // All rules will be scanned, but only those with current priority
1486                     // will be scanned at each step.
1487                     for (var i = 0 ; i < rules.length; i++) {
1488                         var rule = rules[i];
1489                         var priority = rule.getPriority();
1490                         if ((priority < maxPriority) && (priority > nextPriority))
1491                             nextPriority = priority;
1492 
1493                         if (context.control == "nextPriority") {
1494                             continue;
1495                         }
1496                         if (priority == maxPriority) {
1497                             rule.process(context);
1498                             if (context.control == "end") {
1499                                 return context.values.result;
1500                             }
1501                             if (context.control == "abort") {
1502                                 return null;
1503                             }
1504                         }
1505                     }
1506                     if (context.control == "nextPriority") {
1507                         maxPriority = nextPriority;
1508                         context.control = "continue";
1509                         continue;
1510                     }
1511                     if (context.control == "endPriority") {
1512                         return context.result;
1513                     }
1514                     if (context.control == "restart") {
1515                         context.control = "continue";
1516                         maxPriority = firstMaxPriority;
1517                         rules = new Array();
1518                         this.index.getRules(context, rules);
1519                         continue;
1520                     }
1521                     if (context.control == "setPriority") {
1522                         context.control = "continue";
1523                         if (context.priority) {
1524                             maxPriority = context.priority;
1525                         } else {
1526                             maxPriority = nextPriority;
1527                         }
1528                     }
1529                     else {
1530                         context.control = "continue";
1531                         maxPriority = nextPriority;
1532                     }
1533                 }
1534                 return context.getValue("$result");
1535             },
